cmake_minimum_required(VERSION 3.4)
project(example)
set(CMAKE_CXX_STANDART 11)
set(CMAKE_CXX_STANDART_REQUIRED ON)
add_library(ConceptualExample STATIC ${CMAKE_CURRENT_SOURCE_DIR}/ConceptualExample.cpp)
add_library(Vehicles STATIC ${CMAKE_CURRENT_SOURCE_DIR}/Vehicles.cpp)
add_executable(example ${CMAKE_CURRENT_SOURCE_DIR}/Program.cpp)
target_link_libraries(example ConceptualExample)
target_link_libraries(example Vehicles)
